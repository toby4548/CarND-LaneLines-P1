def slope(x1,y1,x2,y2):
    '''
    Return slope of a line
    '''
    return (y2-y1)/(x2-x1)

def ins(x1,y1,x2,y2, position):
    '''
    Return the intersect point of a line with bottom = 0 or top =1
    '''
    if (position == 0):
        return int(((image.shape[0]-y1)/(slope(x1,y1,x2,y2)))+x1)
    elif (position == 1):
        return int((((0.65*image.shape[0])-y1)/(slope(x1,y1,x2,y2)))+x1)
    
def grayscale(img):
    """Applies the Grayscale transform
    This will return an image with only one color channel
    but NOTE: to see the returned image as grayscale
    (assuming your grayscaled image is called 'gray')
    you should call plt.imshow(gray, cmap='gray')"""
    return cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)
    # Or use BGR2GRAY if you read an image with cv2.imread()
    # return cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    
def  RGB2HSV(img):
    '''Applies HSV transfrom for the low light situation
    This will return image which filtered out white and yellow line in HSV form'''
    hsv_img = cv2.cvtColor(img, cv2.COLOR_RGB2HSV)
    sensitivity = 40
    yellow = cv2.inRange(hsv_img, (20, 80, 80), (25, 255, 255))
    white = cv2.inRange(hsv_img, (0, 0, 255-sensitivity), (255, sensitivity, 255))
    gray = cv2.bitwise_or(yellow, white)
    return gray
    
def canny(img, low_threshold, high_threshold):
    """Applies the Canny transform"""
    return cv2.Canny(img, low_threshold, high_threshold)

def gaussian_blur(img, kernel_size):
    """Applies a Gaussian Noise kernel"""
    return cv2.GaussianBlur(img, (kernel_size, kernel_size), 0)

def region_of_interest(img, vertices):
    """
    Applies an image mask.
    
    Only keeps the region of the image defined by the polygon
    formed from `vertices`. The rest of the image is set to black.
    """
    #defining a blank mask to start with
    mask = np.zeros_like(img)   
    
    #defining a 3 channel or 1 channel color to fill the mask with depending on the input image
    if len(img.shape) > 2:
        channel_count = img.shape[2]  # i.e. 3 or 4 depending on your image
        ignore_mask_color = (255,) * channel_count
    else:
        ignore_mask_color = 255
        
    #filling pixels inside the polygon defined by "vertices" with the fill color    
    cv2.fillPoly(mask, vertices, ignore_mask_color)
    
    #returning the image only where mask pixels are nonzero
    masked_image = cv2.bitwise_and(img, mask)
    
    #plt.imshow(masked_image)
    #plt.show()
    return masked_image

def buffer(x_b_L,x_b_R,x_t_L,x_t_R, N=30):
    '''
    Buffer points for average
    '''
    global Buff_x_b_Left
    global Buff_x_b_Right
    global Buff_x_t_Left
    global Buff_x_t_Right
    
    if len(Buff_x_b_Left)<N:
        Buff_x_b_Left=np.append(Buff_x_b_Left,[ x_b_L] )
        Buff_x_b_Right=np.append(Buff_x_b_Right, [x_b_R] )
        Buff_x_t_Left=np.append(Buff_x_t_Left, [x_t_L] )
        Buff_x_t_Right=np.append(Buff_x_t_Right,[ x_t_R] )
    elif len(Buff_x_b_Left) == N:
        for i in range(N-1):
            Buff_x_b_Left[i] = Buff_x_b_Left[i+1]
            Buff_x_b_Right[i] =  Buff_x_b_Right[i+1]
            Buff_x_t_Left[i] = Buff_x_t_Left[i+1]
            Buff_x_t_Right[i] =  Buff_x_t_Right[i+1]
        Buff_x_b_Left[N-1]=x_b_L
        Buff_x_b_Right[N-1] = x_b_R
        Buff_x_t_Left[N-1] = x_t_L
        Buff_x_t_Right[N-1] = x_t_R
        
    #print(Buff_x_b_Left,Buff_x_b_Right,Buff_x_t_Left,Buff_x_t_Right)
        

def draw_lines(img, lines, color=[255, 0, 0], thickness=2):
    """
    NOTE: this is the function you might want to use as a starting point once you want to

    to that shown in P1_example.mp4).  
    
    Think about things like separating line segments by their 
    slope ((y2-y1)/(x2-x1)) to decide which segments are part of the left
    line vs. the right line.  Then, you can average the position of each of 
    the lines and extrapolate to the top and bottom of the lane.
    
    This function draws `lines` with `color` and `thickness`.    
    Lines are drawn on the image inplace (mutates the image).
    If you want to make the lines semi-transparent, think about combining
    this function with the weighted_img() function below
    """
    global Buff_x_b_Left
    global Buff_x_b_Right
    global Buff_x_t_Left
    global Buff_x_t_Right
    
    
    if lines is None:
        
        if Buff_x_b_Left.size ==0:
            return
        else:
            #If Hough Line is not found draw buffer lines average
            cv2.line(img, (int(np.mean(Buff_x_t_Left)), int(0.65* image.shape[0])), (int(np.mean(Buff_x_b_Left)), int(image.shape[0])), color, thickness*5)
            cv2.line(img, (int(np.mean(Buff_x_t_Right)), int(0.65* image.shape[0])), (int(np.mean(Buff_x_b_Right)), int(image.shape[0])), color, thickness*5)
            return
    else:
        Right_Slopes = np.array([])
        Left_Slopes = np.array([])
        Right_Bottoms = np.array([])
        Left_Bottoms =  np.array([])
        Right_Tops = np.array([])
        Left_Tops = np.array([])
            
        for line in lines:
            for x1,y1,x2,y2 in line:
                m=slope(x1,y1,x2,y2)
                
                #print(m)
                #cv2.line(img, (x1, y1), (x2, y2), color, thickness)
                
                #Use slope to devide Lines into two group left and right .
                if (m>0 and (m > 0.3 and m < 1.0)):
                    Right_Slopes = np.append(Right_Slopes,m)
                    Right_Bottoms = np.append(Right_Bottoms,ins(x1,y1,x2,y2,0))
                    Right_Tops = np.append(Right_Tops,ins(x1,y1,x2,y2,1))
                elif (m<0 and (m < -0.3 and m > -1.0)):
                    Left_Slopes=np.append(Left_Slopes,m)
                    Left_Bottoms = np.append(Left_Bottoms,ins(x1,y1,x2,y2,0))
                    Left_Tops = np.append(Left_Tops,ins(x1,y1,x2,y2,1))
                else:
                    if Buff_x_b_Left.size ==0:
                        break
                    else:
                        #If slope is not in the range , draw lines from buffer's average
                        cv2.line(img, (int(np.mean(Buff_x_t_Left)), int(0.65* image.shape[0])), (int(np.mean(Buff_x_b_Left)), int(image.shape[0])), color, thickness*5)
                        cv2.line(img, (int(np.mean(Buff_x_t_Right)), int(0.65* image.shape[0])), (int(np.mean(Buff_x_b_Right)), int(image.shape[0])), color, thickness*5)
                        break
                    
        #print("Right Slope:\n",Right_Slopes,"\nLeft Slope:\n",Left_Slopes)
        #print("Right Bottoms:\n",Right_Bottoms,"\nLeft Bottoms:\n",Left_Bottoms)
        

        #Calculate intersect points for stable lines
        Right_Slopes_avg=np.mean(Right_Slopes)
        Left_Slopes_avg=np.mean(Left_Slopes)
        Right_Bottoms_avg=np.mean(Right_Bottoms)
        Left_Bottoms_avg=np.mean(Left_Bottoms)
        Right_Tops_avg=np.mean(Right_Tops)
        Left_Tops_avg=np.mean(Left_Tops)
    
        #print("Left Bottoms average = ",Left_Bottoms_avg,"\nRight Bottoms average = ",Right_Bottoms_avg)
        #print("Left Tops averege = ",Left_Tops_avg,"\nRight Tops averege = ",Right_Tops_avg)
        
        #If there is average of intersect point is NaN then draw lines from buffer
        if (math.isnan(Left_Bottoms_avg) or math.isnan(Left_Tops_avg) or math.isnan(Right_Bottoms_avg) or math.isnan(Right_Tops_avg)):
            if Buff_x_b_Left.size ==0:
                return
            else:
                cv2.line(img, (int(np.mean(Buff_x_t_Left)), int(0.65* image.shape[0])), (int(np.mean(Buff_x_b_Left)), int(image.shape[0])), color, thickness*5)
                cv2.line(img, (int(np.mean(Buff_x_t_Right)), int(0.65* image.shape[0])), (int(np.mean(Buff_x_b_Right)), int(image.shape[0])), color, thickness*5)

def hough_lines(img, rho, theta, threshold, min_line_len, max_line_gap):
    """
    `img` should be the output of a Canny transform.
        
    Returns an image with hough lines drawn.
    """
    lines = cv2.HoughLinesP(img, rho, theta, threshold, np.array([]), minLineLength=min_line_len, maxLineGap=max_line_gap)
    
    #Print line parameters:
    #print(lines)
    
    line_img = np.zeros((img.shape[0], img.shape[1], 3), dtype=np.uint8)
    draw_lines(line_img, lines)
    return line_img

# Python 3 has support for cool math symbols.

def weighted_img(img, initial_img, α=0.8, β=1., λ=0.):
    """
    `img` is the output of the hough_lines(), An image with lines drawn on it.
    Should be a blank image (all black) with lines drawn on it.
    
    `initial_img` should be the image before any processing.
    
    The result image is computed as follows:
    
    initial_img * α + img * β + λ
    NOTE: initial_img and img must be the same shape!
    """
    return cv2.addWeighted(initial_img, α, img, β, λ)

def image_processing(image):
    
    
    # 1. Reading with HSV
    gray = RGB2HSV(image)    
    #plt.imshow(gray,cmap='gray') # show a grayscaled image
    #plt.show()
    
    '''
    
    #Original: Reading with RGB
    
    # 1. Read in image 
    #image = mpimg.imread('image')
    #print('This image is:', type(image), 'with dimesions:', image.shape)

    #plt.imshow(image)
    #plt.show()

    # 2. Gray Scale transform the image
    gray = grayscale(image)

    #plt.imshow(gray,cmap='gray') # show a grayscaled image
    #plt.show()
     
    '''
     
    # 3. Define kernel size and apply Gaussian Smoothing
    kernel_size = 5
    blur_gray = gaussian_blur(gray,kernel_size)

    #plt.imshow(blur_gray,cmap='gray') # show a grayscaled image
    #plt.show()

    
    # 4. Define low/high threshold for Canny and apply
    low_threshold = 50
    high_threshold = 150
    edges = canny(blur_gray, low_threshold, high_threshold)

    #plt.imshow(edges,cmap='gray') # show a grayscaled image
    #plt.show()

    # 5. Create mask and filter out areas not interested
    imshape=image.shape
    vertices = np.array([[(0.1*imshape[1],0.9*imshape[0]),(0.4*imshape[1], 0.65*imshape[0]), (0.6*imshape[1], 0.65*imshape[0]), (0.9*imshape[1],0.9*imshape[0])]], dtype=np.int32)
    masked_edges = region_of_interest(edges, vertices)

    #plt.imshow(masked_edges,cmap='gray') # show a grayscaled image
    #x= [vertices[0,0,0],vertices[0,1,0],vertices[0,2,0],vertices[0,3,0],vertices[0,0,0]]
    #y= [vertices[0,0,1],vertices[0,1,1],vertices[0,2,1],vertices[0,3,1],vertices[0,0,1]]
    #plt.plot(x, y, 'b--', lw=4)
    #plt.show()

    # 6. Hough Transform
    rho = 2 # distance resolution in pixels of the Hough grid
    theta = np.pi/180 # angular resolution in radians of the Hough grid
    threshold = 15     # minimum number of votes (intersections in Hough grid cell)
    min_line_length =30 #minimum number of pixels making up a line
    max_line_gap = 120#20    # maximum gap in pixels between connectable line segments
    hough_edges = hough_lines(masked_edges, rho, theta, threshold, min_line_length, max_line_gap)
    
    #plt.imshow(hough_edges) # show a grayscaled image
    #plt.show()

    # 7. Weighted image
    lines_edges = weighted_img(hough_edges, image, α=0.8, β=1., λ=0.)
    
    #plt.imshow(lines_edges)
    #plt.show()

    return lines_edges
